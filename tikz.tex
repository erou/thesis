% New command
\newcommand{\drawrec}[3]{
  \ifthenelse{\equal{#1}{1}}{
    \fill (#2,#3) -- ++(1,0) -- ++(0, 1) -- ++(-1, 0) -- cycle;
    }
    {
      \tikzmath{\x = #2; \y = #3; \a = #1;
                \b =\a/2; \xx = \x+\b; \yy = \y+\b; }

      \drawrec{\b}{\x}{\y}
      \drawrec{\b}{\xx}{\y}
      \drawrec{\b}{\x}{\yy}
    }
}
